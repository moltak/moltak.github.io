matplotlib

heat map

import matplotlib.pyplot as plt
import seaborn as sns

plt.figure(figsize=(12, 12)) # 그래프 크기
sns.heatmap(
    df.corr(),
    linewidths=0.1,
    vmax=0.5,
    cmap=plt.cm.gist_heat,
    linecolor='white',
    annot=True)
plt.show()
